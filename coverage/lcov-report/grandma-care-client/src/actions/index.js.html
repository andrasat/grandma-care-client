export const register = user => (
  dispatch => (
    fetch('http://ec2-35-157-203-118.eu-central-1.compute.amazonaws.com/signup', {
      method: 'post',
      body: JSON.stringify(user),
      headers: {

        'Content-Type': 'application/json'

    }).then(res => {

)
